	 EX 7.0 RETORNAR O NOME DA PESSOA, TIPO LOGRADOURO, LOGRADOURO E NÚMERO DE TODOS OS ENDEREÇOS CADASTRADOS NO SISTEMA 
	 R:
	  select a.nome,c.nome as tipo_endereco,b.logradouro,b.numero
      from ms_pessoa a
      inner join ms_pessoaendereco b
      on a.handle = b.pessoa
      inner join ms_tipopessoaendereco c
      on b.tipo = c.handle
	  
	 EX 8.0 SELECIONAR A QUANTIDADE DE PESSOAS CADASTRADAS E O NOME DO TIPO
     SELECT, JOIN, WHERE.
	 R:
	  select b.nome as tipo_pessoa, count(a.handle) as quantidade
      from ms_pessoa a
      inner join ms_tipopessoa b
      on a.tipo = b.handle
      group by b.nome
      order by quantidade desc;
