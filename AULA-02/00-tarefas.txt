-- EXPORTACAO

EX 1.0 EFETUAR A EXPORTACAO DE TODAS AS PESSOAS DO SISTEMA, RETORNANDO AS SEGUINTES COLUNAS.

	"APELIDO DA PESSOA, TIPO, EMAIL"

    	 APELIDO - MS_PESSOA - APELIDO

         TIPO - MS_PESSOA - TIPO - MS_TIPOPESSOA - NOME

         EMAIL - MS_PESSOA - EMAIL
 
EX 2.0 EFETUAR A EXPORTAÇÃO DE TODOS OS ENDEREÇOS DE TODAS AS PESSOAS, ONDE TERÁ AS SEGUINTES COLUNAS.

	"APELIDO DA PESSOA, CEP, TIPOLOGRADOURO, LOGRADOURO E NÚMERO"

	 APELIDO - MS_PESSOA - APELIDO

	 CEP -  MS_PESSOA - ENDERECOFISCAL -> MS_PESSOAENDERECO - CEP -> MS_CEP - CEP

	 TIPOLOGRADOURO - MS_PESSOA - ENDERECOFISCAL -> MS_PESSOAENDERECO - TIPOLOGRADOURO -> MS_TIPOLOGRADOURO - NOME

	 LOGRADOURO -  MS_PESSOA - ENDERECOFISCAL -> MS_PESSOAENDERECO - LOGRADOURO 

	 NUMERO - MS_PESSOA - ENDERECOFISCAL -> MS_PESSOAENDERECO - NUMERO
 
 
EX 3.0 CRIAR UM LAYOUT DE EDI JSON PARA QUE SEJA EXPORTADO DADOS DE UMA ÚNICA PESSOA COM AS SEGUINTES INFORMAÇÕES
 
{

    "Pessoa": {

        "Tipo": "x",

        "Nome": "xxx",

        "Cnpj_cpf": "xxx"

    }

}

----- Logica do layout -----

{ --objeto

    "Pessoa": { --objeto

        "Tipo": "x", --atributo

        "Nome": "xxx", --atributo

        "Cnpj_cpf": "xxx" --atributo

    }

}
